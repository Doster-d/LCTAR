416class FrameAccurateVideoRecording extends VideoRecording {

427 async captureFrame(frame) {
428 try {
429 this.isCapturingFrame = true;
430 if (this.firstFrame === null) this.firstFrame = frame;
431 await this.canvasSource.add(
432 this.toTime(frame) - this.toTime(this.firstFrame),
433 this.toTime(1)
434 );
435 this.lastCapturedFrame = frame;
436 if (this.toTime(frame - this.firstFrame + 1) >= this.duration) {
437 await this.stop();
438 }
439 } catch (err) {
440 await this.cancelWithReason(err);
441 } finally {
442 this.isCapturingFrame = false;
443 }
