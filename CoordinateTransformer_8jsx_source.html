38 calculateOptimalOrigin() {
39 if (this.anchors.size === 0) return;
40
41 let weightedSum = new Vector3();
42 let totalWeight = 0;
43
44 this.anchors.forEach((anchor, id) => {
45 const weight = anchor.confidence || 1.0;
46 weightedSum.add(anchor.position.clone().multiplyScalar(weight));
47 totalWeight += weight;
48 });
49
50 if (totalWeight > 0) {
51 this.origin.copy(weightedSum.divideScalar(totalWeight));
52 }
53 }

58 calculateWorldRotation() {
59 if (this.anchors.size === 0) return;
60
61 // Используем первую якорь как основу для вычисления вращения
62 const firstAnchor = this.anchors.values().next().value;
63 if (!firstAnchor) return;
64
65 // Создаем систему координат где Z направлена по нормали якоря
66 const zAxis = firstAnchor.normal.clone().normalize();
67 const xAxis = new Vector3(1, 0, 0);
68 const yAxis = new Vector3().crossVectors(zAxis, xAxis).normalize();
69 xAxis.crossVectors(yAxis, zAxis).normalize();
70
71 this.rotation.setFromRotationMatrix(
72 new Matrix4().makeBasis(xAxis, yAxis, zAxis)
73 );
74 }
75
79 updateWorldMatrix() {
80 this.worldMatrix.makeRotationFromQuaternion(this.rotation);
81 this.worldMatrix.setPosition(this.origin);

88export class StabilizationFilter {
89 constructor(type = 'exponential', config = {}) {
90 this.type = type;
91 this.config = {
92 alpha: 0.1, // Коэффициент сглаживания для экспоненциального фильтра
93 kalmanProcessNoise: 0.01, // Процессный шум для Калмана
94 kalmanMeasurementNoise: 0.1, // Измерительный шум для Калмана
95 medianWindowSize: 5, // Размер окна для медианного фильтра
96 ...config
97 };
98
99 this.reset();
100 }
101
105 reset() {
106 this.history = [];
107 this.previousValue = null;
108
109 // Сбрасываем кэш медианного фильтра
110 if (this.medianSortCache) {
111 this.medianSortCache.fill(0);
112 }
113
114 // Состояние Калмановского фильтра
115 this.kalman = {
116 x: 0, // Предсказанное значение
117 P: 1, // Ковариация ошибки предсказания
118 K: 0 // Коэффициент усиления Калмана
119 };
120 }
121
127 apply(value) {
128 if (this.previousValue === null) {
129 this.previousValue = value;
130 return value;
131 }
132
133 switch (this.type) {
134 case 'kalman':
135 return this.applyKalmanFilter(value);
136 case 'median':
137 return this.applyMedianFilter(value);
138 case 'exponential':
139 default:
140 return this.applyExponentialFilter(value);
141 }
142 }
143
147 applyExponentialFilter(value) {
148 const alpha = this.config.alpha;
149 const filtered = alpha * value + (1 - alpha) * this.previousValue;
150 this.previousValue = filtered;
151 return filtered;
152 }
153
157 applyKalmanFilter(value) {
158 const { kalmanProcessNoise, kalmanMeasurementNoise } = this.config;
159
160 // Предсказание
161 const P_pred = this.kalman.P + kalmanProcessNoise;
162
163 // Обновление
164 this.kalman.K = P_pred / (P_pred + kalmanMeasurementNoise);
165 this.kalman.x = this.kalman.x + this.kalman.K * (value - this.kalman.x);
166 this.kalman.P = (1 - this.kalman.K) * P_pred;
167
168 this.previousValue = this.kalman.x;
169 return this.kalman.x;
170 }
171
175 applyMedianFilter(value) {
176 const startTime = performance.now();
177
178 this.history.push(value);
179 if (this.history.length > this.config.medianWindowSize) {
180 this.history.shift();
181 }
182
183 if (this.history.length < 3) {
184 this.previousValue = value;
185 this.recordPerformance(startTime, 'filter');
186 return value;
187 }
188
189 // Используем кэшированный массив для сортировки вместо создания нового
190 if (!this.medianSortCache) {
191 this.medianSortCache = new Float32Array(this.config.medianWindowSize);
192 }
193
194 // Копируем данные в кэшированный массив
195 const len = this.history.length;
196 for (let i = 0; i < len; i++) {
197 this.medianSortCache[i] = this.history[i];
198 }
199
200 // Сортируем кэшированный массив
201 this.medianSortCache.sort((a, b) => a - b);
202
203 const mid = Math.floor(len / 2);
204 const filtered = len % 2 === 0
205 ? (this.medianSortCache[mid - 1] + this.medianSortCache[mid]) / 2
206 : this.medianSortCache[mid];
207
208 this.previousValue = filtered;
209 this.recordPerformance(startTime, 'filter');
210 return filtered;
211 }

674 calculateDetectionConfidence(detection) {
675 // Уверенность на основе качества детекции
676 const baseConfidence = detection.rawDetection ? 0.8 : 0.5;
677
678 // Модификатор на основе расстояния до маркера
679 const distance = Math.sqrt(
680 detection.position[0] ** 2 +
681 detection.position[1] ** 2 +
682 detection.position[2] ** 2
683 );
684
685 const distanceModifier = Math.max(0.1, Math.min(1.0, 1.0 / (1.0 + distance * 2)));
686
687 return baseConfidence * distanceModifier;
688 }

714 applyStabilization(timestamp) {
715
716 const deltaTime = (timestamp - this.currentTransform.timestamp) / 1000.0; // в секундах
717 if (deltaTime <= 0) {
718 return;
719 }
720
721 // Вычисляем адаптивную скорость лерпинга
722 const lerpSpeed = this.calculateAdaptiveLerpSpeed(deltaTime);
723
724 // Проверяем dead zone для позиции
725 const positionDiff = this.targetTransform.position.distanceTo(this.currentTransform.position);
726 if (positionDiff > this.config.positionDeadZone) {
727 const filteredPosition = this.applyPositionFiltering(this.targetTransform.position);
728 this.currentTransform.position.lerp(filteredPosition, lerpSpeed);
729 }
730
731 // Проверяем dead zone для вращения
732 const rotationDiff = this.targetTransform.rotation.angleTo(this.currentTransform.rotation);
733 if (rotationDiff > this.config.rotationDeadZone) {
734 const filteredRotation = this.applyRotationFiltering(this.targetTransform.rotation);
735 this.currentTransform.rotation.slerp(filteredRotation, lerpSpeed);
736 }
737
738 const oldConfidence = this.currentTransform.confidence;
739 this.currentTransform.confidence = this.targetTransform.confidence;
740 this.currentTransform.timestamp = timestamp;
741
742
743 // Добавляем в историю для предиктивной стабилизации
744 this.addToHistory(this.currentTransform);
745 }

750 calculateAdaptiveLerpSpeed(deltaTime) {
751 const confidenceFactor = this.targetTransform.confidence;
752
753 // Базовая скорость на основе уверенности
