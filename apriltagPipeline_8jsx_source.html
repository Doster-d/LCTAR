317 if (!detections || detections.length === 0) {

344 calculateDetectionConfidence(detection) {
345 // Базовая уверенность
346 const baseConfidence = detection.rawDetection ? 0.8 : 0.5;
347
348 // Модификатор на основе расстояния до маркера
349 const distance = Math.sqrt(
350 detection.position[0] ** 2 +
351 detection.position[1] ** 2 +
352 detection.position[2] ** 2
353 );
354
355 const distanceModifier = Math.max(0.1, Math.min(1.0, 1.0 / (1.0 + distance * 2)));
356
357 return baseConfidence * distanceModifier;
358 }

366 applyRecoveryStrategies(currentDetections, timestamp) {
367 const timeSinceLastDetection = timestamp - this.lastDetectionTime;
368
369 // Если есть текущие детекции, сохраняем их для будущего восстановления
370 if (currentDetections.length > 0) {
371 currentDetections.forEach(detection => {
372 this.recoveryStrategies.lastKnownTransforms.set(detection.id, {
373 ...detection,
374 lastSeen: timestamp
375 });
376 });
377 return currentDetections;
378 }
379
380 // Если нет детекций, применяем стратегии восстановления
381 const recoveredDetections = [];
382
383 if (this.recoveryStrategies.fallbackToLastKnown) {
384 // Fallback к последним известным трансформациям
385 this.recoveryStrategies.lastKnownTransforms.forEach((lastKnown, id) => {
386 const timeSinceLastSeen = timestamp - lastKnown.lastSeen;
387
388 // Не используем трансформации старше максимального времени предикции
389 if (timeSinceLastSeen < this.recoveryStrategies.maxPredictionTime) {
390 const decayFactor = 1.0 - (timeSinceLastSeen / this.recoveryStrategies.maxPredictionTime);
391
392 recoveredDetections.push({
393 ...lastKnown,
394 confidence: lastKnown.confidence * decayFactor * 0.5, // Уменьшаем уверенность
395 recovered: true,
396 recoveryMethod: 'fallback'
397 });
398 }
399 });
400 }
401
402 return recoveredDetections;
403 }
