474 convertImageDataToGrayscale(imageData) {
475 const data = imageData.data;
476 const grayscale = new Uint8Array(data.length / 4);
477
478 for (let i = 0; i < data.length; i += 4) {
479 // Convert RGB to grayscale using standard formula
480 const gray = Math.round(0.299 * data[i] + 0.587 * data[i + 1] + 0.114 * data[i + 2]);
481 grayscale[i / 4] = gray;
482 }
483
484 return grayscale;
485 }
