93// Функция получения CSRF токена из cookies
94const getCsrfToken = () => {
95 const name = 'csrftoken';
96 let cookieValue = null;
97 if (document.cookie && document.cookie !== '') {
98 const cookies = document.cookie.split(';');
99 for (let i = 0; i < cookies.length; i++) {
100 const cookie = cookies[i].trim();
101 if (cookie.substring(0, name.length + 1) === (name + '=')) {
102 cookieValue = decodeURIComponent(cookie.substring(name.length + 1));
103 break;
104 }
105 }
106 }
107 return cookieValue;
108};

128// Основная функция запроса
129export async function request(path, { method = 'GET', body, headers, baseUrl, ...rest } = {}) {
130 const init = {
131 method,
132 ...rest,
133 };
134
135 const resolvedBaseUrl = typeof baseUrl === 'string'
136 ? normalizeBaseUrl(baseUrl)
137 : apiState.baseUrl;
138
139 // Создаем базовые заголовки
140 const requestHeaders = {
141 ...(body !== undefined ? { 'Content-Type': 'application/json' } : {}),
142 ...(headers || {}),
143 };
144
145 // Добавляем CSRF токен для не-GET методов
146 if (method !== 'GET') {
147 const csrfToken = getCsrfToken();
148 if (csrfToken) {
149 requestHeaders['X-CSRFToken'] = csrfToken;
150 }
151 }
152
153 init.headers = requestHeaders;
154
155 if (body !== undefined) {
156 init.body = JSON.stringify(body);
157 }
158
159 let response;
160 try {
161 response = await fetch(buildUrl(path, resolvedBaseUrl), init);
162 } catch (error) {
163 const networkError = new Error('Network error');
164 networkError.cause = error;
165 throw networkError;
166 }
167
168 const payload = await parseResponseBody(response);
169
170 if (!response.ok) {
171 const detail = payload?.detail || payload?.message || payload;
172 const error = new Error(typeof detail === 'string' ? detail : 'Request failed');
173 error.status = response.status;
174 error.payload = payload;
175 throw error;
176 }
177
178 return payload;
179}
