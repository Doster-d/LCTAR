8import React, { forwardRef, useEffect, useMemo, useRef } from 'react'
9import { useGraph } from '@react-three/fiber'
10import { useGLTF, useAnimations } from '@react-three/drei'
11import { SkeletonUtils } from 'three-stdlib'

19export const Model = forwardRef((props, ref) => {
20 const group = useRef()
21
22 // Загрузка моделей
23 const { scene: trainScene, animations: trainAnimations } = useGLTF('./models/Train-transformed.glb')
24 const { scene: orangeboxScene, animations: orangeboxAnimations } = useGLTF('models/orangebox-transformed.glb')
25
26 // Клонирование сцен для переиспользования
27 const clonedTrainScene = useMemo(() => SkeletonUtils.clone(trainScene), [trainScene])
28 const clonedOrangeboxScene = useMemo(() => SkeletonUtils.clone(orangeboxScene), [orangeboxScene])
29
30 // Получение nodes и materials для обеих моделей
31 const { nodes: trainNodes, materials: trainMaterials } = useGraph(clonedTrainScene)
32 const { nodes: orangeboxNodes, materials: orangeboxMaterials } = useGraph(clonedOrangeboxScene)
33
34 // Анимации для обеих моделей
35 const { actions: trainActions } = useAnimations(trainAnimations, group)
36 const { actions: orangeboxActions } = useAnimations(orangeboxAnimations, group)
37
38 // Ref forwarding
39 useEffect(() => {
40 if (!ref) return
41 if (typeof ref === 'function') ref(group.current)
42 else ref.current = group.current
43 }, [ref])
44
45 // Настройка анимаций для поезда
46 useEffect(() => {
47 if (!group.current) return
48 const animationsList = Array.isArray(trainAnimations) ? trainAnimations : []
49 group.current.userData.animations = animationsList
50 group.current.animations = animationsList
51 }, [trainAnimations])
52
53 // Запуск анимаций поезда
54 useEffect(() => {
55 if (!trainActions) return
56 const actionList = Object.values(trainActions)
57 if (!actionList.length) return
58 actionList.forEach(action => action.reset().play())
59 }, [trainActions])
60
61 // Запуск анимаций коробки с апельсинами
62 useEffect(() => {
63 if (!orangeboxActions) return
64 const actionList = Object.values(orangeboxActions)
65 if (!actionList.length) return
66 actionList.forEach(action => action.reset().play())
67 }, [orangeboxActions])
68
69 // Получение узлов модели поезда
70 const trainBodyNode = trainNodes['Train-Mat.2'] ?? trainNodes['Train-Mat2'] ?? trainNodes['Train_Mat.2'] ?? trainNodes['Train_Mat2']
71 const trainWagonNode = trainNodes['Train-Wagon']
72 const trainWheelsNode = trainNodes['Train-Wheels']
73 const railsNode = trainNodes.Rails
74
75 // Получение узлов модели коробки с апельсинами
76 const orangesBoxNode = orangeboxNodes.Oranges_Box
77 const orangeNode = orangeboxNodes.Orange
78
79 if (!trainBodyNode) {
80 console.error('Train model mesh "Train-Mat.2" not found. Available nodes:', Object.keys(trainNodes))
81 }
82
83 return (
84 <group ref={group} {...props} dispose={null}>
85 <group>
86 <group name="Null">
87 {/* Модель поезда */}
88 <group name="Train" position={[0, 0.26, 0]} rotation={[0, Math.PI / 2, 0]}>
89 {trainBodyNode && (
90 <mesh
91 name={trainBodyNode.name ?? 'Train-Mat.2'}
92 geometry={trainBodyNode.geometry}
93 material={trainMaterials['Mat.2']}
94 />
95 )}
96 {trainWagonNode && (
97 <mesh
98 name={trainWagonNode.name ?? 'Train-Wagon'}
99 geometry={trainWagonNode.geometry}
100 material={trainMaterials.Wagon}
101 />
102 )}
103 {trainWheelsNode && (
104 <mesh
105 name={trainWheelsNode.name ?? 'Train-Wheels'}
106 geometry={trainWheelsNode.geometry}
107 material={trainMaterials.Wheels}
108 />
109 )}
110
111
112 {/* Отдельный апельсин рядом с поездом */}
113 {orangeNode && (
114 <mesh
115 name="Orange_near_Train"
116 geometry={orangeNode.geometry}
117 material={orangeboxMaterials.Mat}
118 position={[-0.068 + 0, 0.123 + 0.26, 0.14 + 0]}
119 />
120 )}
121 </group>

124 {orangesBoxNode && (
125 <group name="Orangebox_Stationary" position={[0.25, 0, 0]}>

127 <mesh name="Orange6" geometry={orangesBoxNode.children[0]?.geometry} material={orangeboxMaterials.Mat} position={[0.017, 0.023, -0.076]} rotation={[0.326, 0, -0.544]} />
128 <mesh name="Orange4" geometry={orangesBoxNode.children[1]?.geometry} material={orangeboxMaterials.Mat} position={[-0.024, 0.029, 0.018]} rotation={[1.126, 0.001, -0.045]} />
