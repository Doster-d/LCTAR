74export function bestFitPointFromRays(rays) {
75 if (!Array.isArray(rays) || rays.length === 0) {
76 return null;
77 }
78
79 if (rays.length === 1) {
80 // Для одного луча возвращаем точку на расстоянии 1 от origin
81 return rays[0].origin.clone().add(
82 rays[0].direction.clone().normalize()
83 );
84 }
85
86 // Метод наименьших квадратов для нахождения точки, минимизирующей сумму расстояний до всех лучей
87 const origins = rays.map(ray => ray.origin);
88 const directions = rays.map(ray => ray.direction.clone().normalize());
89
90 // Вычисляем центр масс начал лучей
91 const centroid = new THREE.Vector3();
92 origins.forEach(origin => centroid.add(origin));
93 centroid.divideScalar(origins.length);
94
95 // Итеративное уточнение точки пересечения
96 let bestPoint = centroid.clone();
97 const maxIterations = 100;
98 const tolerance = 1e-6;
99
100 for (let iteration = 0; iteration < maxIterations; iteration++) {
101 let totalDirection = new THREE.Vector3();
102 let totalWeight = 0;
103
104 for (let i = 0; i < rays.length; i++) {
105 const origin = origins[i];
106 const direction = directions[i];
107
108 // Вектор от начала луча к текущей точке
109 const toPoint = bestPoint.clone().sub(origin);
110
111 // Проекция вектора на направление луча
112 const projection = toPoint.dot(direction);
113
114 // Направление коррекции
115 const correction = toPoint.clone().sub(
116 direction.clone().multiplyScalar(projection)
117 );
118
119 // Взвешиваем коррекцию
120 const weight = 1.0 / (1.0 + correction.lengthSq());
121 totalDirection.add(correction.multiplyScalar(weight));
122 totalWeight += weight;
123 }
124
125 if (totalWeight > 0) {
126 totalDirection.divideScalar(totalWeight);
127 bestPoint.sub(totalDirection);
128
129 // Проверяем сходимость
130 if (totalDirection.lengthSq() < tolerance * tolerance) {
131 break;
132 }
133 }
134 }
135
136 return bestPoint;
137}

144export function clampQuaternion(quaternion) {
145 // Проверяем тип входного параметра и создаем корректный кватернион при необходимости
146 let quat;
147 if (quaternion instanceof THREE.Quaternion) {
148 quat = quaternion;
149 } else if (typeof quaternion === 'object' && quaternion !== null && 'w' in quaternion && 'x' in quaternion && 'y' in quaternion && 'z' in quaternion) {
150 // Создаем кватернион из объекта с компонентами
151 quat = new THREE.Quaternion(quaternion.x, quaternion.y, quaternion.z, quaternion.w);
152 } else {
153 return new THREE.Quaternion();
154 }
155
156 // Выполняем операции clone и normalize с проверкой результата
157 let normalized;
158 try {
159 const cloned = quat.clone();
160 if (typeof cloned.normalize === 'function') {
161 normalized = cloned.normalize();
162 } else {
163 normalized = cloned;
164 }
165 } catch (error) {
166 return new THREE.Quaternion();
167 }
168
169 // Проверяем, что normalized является корректным THREE.Quaternion объектом
170 if (!(normalized instanceof THREE.Quaternion)) {
171 return new THREE.Quaternion();
172 }
173
174 // Убеждаемся, что кватернион в правильном диапазоне
175 // Избегаем кватернионов с отрицательной скалярной частью для непрерывности
176 if (normalized.w < 0 && typeof normalized.multiplyScalar === 'function') {
177 normalized.multiplyScalar(-1);
178 }

216function softenAngle(angle, deadZone, softZone) {
217 const absAngle = Math.abs(angle);
218
219 if (absAngle <= deadZone) {
220 // В зоне мертвого угла - обнуляем угол
221 return 0;
222 } else if (absAngle <= softZone) {
223 // В зоне мягкого перехода - применяем нелинейную фильтрацию
224 const normalizedAngle = (absAngle - deadZone) / (softZone - deadZone);
225 const filteredAngle = Math.sin(normalizedAngle * Math.PI / 2) * (absAngle - deadZone);
226 return Math.sign(angle) * (deadZone + filteredAngle);
227 }
228
229 // За пределами зоны - возвращаем угол без изменений
230 return angle;
231}
